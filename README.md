Configuración adicional:
  -sublime:
     Preferences -> Package Settings -> SublimeGit -> Settings — User
     {
       "git_executables": {
         "git": ["C:\\Program Files\\Git\\bin\\git.exe"]
       }
      }
  
  
  -consola git:
    git config --global user.email "you@example.com"
    hit enter
    git config --global user.name "Your Name"
    hit enter
